```
clc;
close all;
clear all;

t=-10:0.01:10;
T=4
fm=1/T;
x=cos(2*pi*fm*t);
subplot(2,2,1);
plot(t,x);
xlabel('time');
ylabel('x(t)');
title('continous time signal')
grid;
n1=-10:1:10
fs1=1.6*fm;
fs2=2*fm;
fs3=8*fm;
x1=cos(2*pi*fm/fs1*n1);
subplot(2,2,2);
stem(n1,x1);
xlabel('time');
ylabel('x(n)');
title('discrete time signal with fs<2fm');
hold on
subplot(2,2,2);
plot(n1,x1);
grid on;
x2=cos(2*pi*fm/fs2*n1);
subplot(2,2,3);
stem(n1,x2);
xlabel('time');
ylabel('x(n)');
title('discrete time signal with fs=2fm');
hold on
subplot(2,2,3);
plot(n1,x2);
grid on;
x3= cos(2*pi*fm/fs3*n1);
subplot(2,2,4);
stem(n1,x3);
xlabel('time');
ylabel('x(n)');
title('discrete time signal with fs>2fm');
hold on;
subplot(2,2,4);
plot(n1,x3);
grid on;
```
